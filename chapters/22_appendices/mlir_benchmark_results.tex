        cd llvm-project-benchmarks/build
        cmake -G Ninja ../llvm \
           -DLLVM_ENABLE_PROJECTS=mlir \
           -DLLVM_TARGETS_TO_BUILD="host" \
           -DLLVM_ENABLE_BENCHMARKS=ON \
           -DCMAKE_BUILD_TYPE=Release \
